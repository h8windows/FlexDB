def authorized?(permission, thing, &block)
  block.call if can?(permission.to_sym, thing) || current_user.try(:admin?)
  nil
end

def title(*parts)
  unless parts.empty?
    content_for :title do
      (parts << "FlexDB") .join(" - ") unless parts.empty?
    end
  end
end
